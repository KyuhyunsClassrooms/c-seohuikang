# 1. 프로그램 설계서 (5점)

## 1. 프로젝트 주제
스트레칭 추천 프로그램
## 2. 핵심 기능 목록
(구현할 기능들을 순서대로 나열하세요.)
1.  사용자 입력 기능(사용자의 신체나 아픈 부위를 입력받는 기능)
2.  스트레칭 데이터 준비 기능(각 부위별로 추천할 스트레칭 문장을 배열로 저장)
3.  랜덤 스트레칭 선택 기능(사용자가 입력한 부위에 따라 해당 배열에서 무작위로 스트레칭 선택)
4.  결과 출력 기능(사용자의 입력과 함께 추천 결과를 화면에 출력, 입력값에 따라 다른 부위 출력)

## 3. 데이터 구조 설계
(프로그램에 필요한 핵심 변수, 배열을 설계하세요. 필요하면 구조체(struct)를 사용할 수도 있습니다.)

**방법 1: 배열 사용 (기본)**
```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    // 부위별 스트레칭 이름을 2차원 문자열 배열로 저장
    char shoulder[3][30] = {"어깨 돌리기", "벽 밀기 스트레칭", "팔 뻗기"};
    char waist[3][30]    = {"고양이 자세", "옆구리 늘리기", "브리지"};
    char neck[3][30]     = {"목 옆으로 기울이기", "고개 돌리기", "턱 당기기"};
    char leg[3][30]      = {"햄스트링 스트레칭", "종아리 늘리기", "스쿼트"};

    // 부위 목록 배열 (입력 검증용)
    char parts[4][10] = {"어깨", "허리", "목", "다리"};

    // 사용자 입력 및 랜덤 초기화
    char input[20];
    srand(time(NULL));

    printf("어느 부위가 불편한가요? (어깨/허리/목/다리): ");
    scanf("%s", input);

    // 입력에 따라 해당 부위 배열에서 스트레칭 추천
    int index = rand() % 3; // 각 부위별 스트레칭 개수는 3개

    printf("\n=== 스트레칭 추천 결과 ===\n");

    if (strcmp(input, "어깨") == 0)
        printf("추천 스트레칭: %s\n", shoulder[index]);
    else if (strcmp(input, "허리") == 0)
        printf("추천 스트레칭: %s\n", waist[index]);
    else if (strcmp(input, "목") == 0)
        printf("추천 스트레칭: %s\n", neck[index]);
    else if (strcmp(input, "다리") == 0)
        printf("추천 스트레칭: %s\n", leg[index]);
    else
        printf("입력하신 부위를 찾을 수 없습니다.\n");

    printf("스트레칭 전후에는 충분히 몸을 풀어주세요!\n");

    return 0;
}
